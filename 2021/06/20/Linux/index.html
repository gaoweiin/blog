Linux

文件操作:

mv 源 目标

cp 源 目标

touch 文件名

mkdir 文件夹名

ls 显示当前目录下未隐藏的文件

tar -czvf files.tar.gz .打包

tar -xzvf files.tar.gz -C a解包

tar -cvf打包

tar -zcvf压缩打包

tar -xvf解包

tar -xzvf解压缩

NFS:

服务器端操作

yum install rpcbind nfs-utils -y

mkdir /home/wei/nfs

chmod 777 -R /home/wei/nfs

vim /etc/sysconfig/network-scripts/ifcfg-ens33

ifdown ens33

ifup ens33

systemctl start rpcbind

systemctl start nfs-server.service

vim /etc/exports

exportfs -rv

客户端操作

mount -t nfs 192.168.83.212:/home/wei/nfs /home/wei/nfsc

Samba;

服务器端操作

setenforce 0 关SELinux

yum install samba

设置共享目录

mkdir /usr/share/smbs

chmod 777 -R /usr/share/smbs

semanage fcontext -at samba_share_t "/usr/share/smbs(/.*)?"

restorecon /usr/share/smbs

vim /etc/samba/smb.conf

testparm

systemctl restart smb

systemctl restart nmb
